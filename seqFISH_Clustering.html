plist <- lapply(c("louv_expr", "louv_spat"), function(clus_algo){
  p <- ggplot(df, aes_string(x = "x", y = "y", col = clus_algo)) +
    geom_point(size = 1.5) +
    # guides(color = FALSE) +
    labs(title = clus_algo, col = "clusters") +
    theme(legend.position = "bottom")
  return(p)
})
